    import java.io.*;
    import java.awt.*;
    importjava.applet.*;
    import java.util.*;
    public class DDA extendsApplet {
    public void paint(Graphics g) {
        double dx, dy, steps, x, y, k;
        double xc, yc;
        double x1 = 200, y1 = 500, x2 = 600, y2 = 200;
        dx = x2 - x1;
        dy = y2 - y1;
        if (Math.abs(dx) > Math.abs(dy))
            steps = Math.abs(dx);
        else
            steps = Math.abs(dy);
        xc = (dx / steps);
        yc = (dy / steps);
        x = x1;
        y = y1;
        g.fillOval(200, 500, 5, 5);
        for (k = 1; k <= steps; k++) {
            x = x + xc;
            y = y + yc;
            g.fillOval((int) x, (int) y, 5, 5);
        }
    }
}

    public class circle extends Applet {
    public void paint(Graphics g) {
        int xc, yc, r;
        xc = 125;
        yc = 125;
        r = 60;
        int x = 0;
        int y = r;
        int p = 3 - 2 * r;
        do {
            if (p < 0)
                p = p + 4 * x + 6;
            else {
                p = p + 4 * (x - y) + 10;
                y = y - 1;
            }
            x = x + 1;
            g.drawLine(xc + x, yc + y, xc + x, yc + y);
            g.drawLine(x + xc, yc - y, xc + x, yc - y);
            g.drawLine(xc - x, yc + y, xc - x, yc + y);
            g.drawLine(xc - x, yc - y, xc - x, yc - y);
            g.drawLine(xc + y, yc + x, xc + y, yc + x);
            g.drawLine(xc + y, yc - x, xc + y, yc - x);
            g.drawLine(xc - y, yc + x, xc - y, yc + x);
            g.drawLine(xc - y, yc - x, xc - y, yc - x);
        }
        while (x < y);
    }
}

    /*(applet code="tilecase" width=900 height=900>(/applet>*/  
    
    public class tilecase extends Applet implements ActionListener { 
        Image img; 
        int w = 100, h = 100, x, y, c; 
        Button b1 = new Button("tile"); 
        Button b2 = new Button("cascade"); 
        
        public void init() { 
            img = getImage(getCodeBase(), "1.jpg"); 
            b1.addActionListener(this); 
            add(b1); 
            b2.addActionListener(this); 
            add(b2); 
        } 
        
        public void actionPerformed(ActionEvent ae) { 
            if (ae.getActionCommand() == "tile") 
                c = 1; 
            if (ae.getActionCommand() == "cascade") 
                c = 2; 
            repaint(); 
        } 
        
        public void paint(Graphics r) { 
            x = 0; 
            y = 0; 
            
            if (c == 1) { 
                while (y < getHeight()) { 
                    r.drawImage(img, x, y, this); 
                    x += w; 
                    if (x > getWidth()) { 
                        x = 0; 
                        y += h; 
                    } 
                } 
            } 
            
            if (c == 2) { 
                for (int i = 0; i <= 8; i++) { 
                    r.drawImage(img, x, y, this); 
                    x += 30; 
                    y += 30; 
                } 
            } 
        } 
    }

public class size extends Applet { 
    Image img; 
    int w, h, x, y; 
    
    public void init() { 
        w = 500; 
        h = 500; 
        x = 0; 
        y = 0; 
        img = getImage(getDocumentBase(), "1.jpg"); 
    } 
    
    public void paint(Graphics g) { 
        g.drawImage(img, x, y, w, h, this); 
        
        try { 
            Thread.sleep(100); 
        } catch (InterruptedException ie) {} 
        
        if (w > 0) 
            repaint();

        y += 5; 
        h -= 5; 
        w -= 5; 
    }
